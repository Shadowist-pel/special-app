Основные команды
git status						# Отображает информацию об изменениях файлов в рабочей директории и о состоянии каждого такого файла
git init						# При помощи команды создали репозиторий в папке проекта (появилась папка git, в которой сохраняются данные репозитория)
git add .						# Добавляет файлы и папки в репозиторий
git add <file(s)>				# Добавление в индекс выбранный файл <имя файла>
git commit -m 'comment'			# Сохраняет изменения в репозитории (-m 'комментарий')
git show						# Показывает изменения последнего коммита. Проверяет что в коммит попали только те изменения, которые добавлены в индекс (хэш автор комментарии дату время)
git log							# Просматриваем всю историю изменений, все коммиты что были в репозитории (хэш автор комментарии дату время)
